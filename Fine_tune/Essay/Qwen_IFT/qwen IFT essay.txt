# ============================================================
# IFT Fine-tuning (Instruction Fine-Tuning) — Essays Dataset
# Model: Qwen2.5-3B-Instruct + QLoRA (ChatML Template)
# Output: Training logs + Accuracy/Precision/Recall/F1 (IFT Named)
# Platform: Kaggle GPU T4
# ============================================================

# !pip install transformers trl peft bitsandbytes accelerate -q

import os
import gc
import json
import torch
import time
import pandas as pd
import numpy as np
from datasets import Dataset
from sklearn.model_selection import train_test_split
from sklearn.metrics import (
    accuracy_score, precision_score,
    recall_score, f1_score, classification_report
)
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    BitsAndBytesConfig,
    TrainerCallback,
)
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
from trl import SFTTrainer, SFTConfig

# ============================================================
# 1. CONFIGURATION
# ============================================================
MODEL_NAME = "Qwen/Qwen2.5-3B-Instruct"   # Sử dụng bản Instruct để làm IFT
DATA_PATH  = "/kaggle/input/datasets/fakekin/datn-final/essays_clean.csv"
OUTPUT_DIR = "/kaggle/working/outputs_ift_essays"   # Đã đổi tên sang ift
TRAITS     = ["cEXT", "cNEU", "cAGR", "cCON", "cOPN"]
MAX_LENGTH = 2048
SEED       = 42

print(f"🚀 MODE: Instruction Fine-Tuning (IFT)")
print(f"Model: {MODEL_NAME}")
print(f"GPU: {torch.cuda.get_device_name(0) if torch.cuda.is_available() else 'CPU'}")
print(f"VRAM: {torch.cuda.get_device_properties(0).total_memory/1e9:.1f}GB")

# ============================================================
# 2. LOAD & SPLIT RAW DATA
# ============================================================
df = pd.read_csv(DATA_PATH)
df = df[["text"] + TRAITS].dropna().reset_index(drop=True)
print(f"\nTotal samples in dataset: {len(df)}")

# Tách tập Test độc lập (20%) - Tuyệt đối không tham gia vào quá trình Train/Eval
train_df, test_df = train_test_split(df, test_size=0.2, random_state=SEED)
# Tách tập Validation từ tập Train (10% của tập Train) để tính Validation Loss
train_df, val_df  = train_test_split(train_df, test_size=0.1, random_state=SEED)
print(f"Train set: {len(train_df)} | Val set: {len(val_df)} | Test set: {len(test_df)}")

# Lưu test set riêng ra bộ nhớ
test_df.to_csv("/kaggle/working/test_essays.csv", index=False)
print("Saved test_essays.csv successfully.")

# ============================================================
# 3. LOAD MODEL & TOKENIZER + QLoRA (Đưa lên trước để lấy Tokenizer)
# ============================================================
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.float16,
    bnb_4bit_use_double_quant=True,
)

model = AutoModelForCausalLM.from_pretrained(
    MODEL_NAME,
    quantization_config=bnb_config,
    device_map="auto",
)
tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME)
tokenizer.pad_token    = tokenizer.eos_token
tokenizer.padding_side = "right" # Cực kỳ quan trọng đối với dòng CausalLM khi train
model.config.use_cache = False

# Cấu hình Adapter Peft LoRA
model = prepare_model_for_kbit_training(model)
model = get_peft_model(model, LoraConfig(
    r=16, lora_alpha=32,
    target_modules=["q_proj","k_proj","v_proj","o_proj",
                    "gate_proj","up_proj","down_proj"],
    lora_dropout=0.05,
    bias="none",
    task_type="CAUSAL_LM",
))
print("\n=== Trainable Parameters ===")
model.print_trainable_parameters()

# ============================================================
# 4. IFT DATA FORMATTING (Sử dụng ChatML Template)
# ============================================================
def format_ift(examples):
    texts = []
    for i in range(len(examples["text"])):
        label = {t: examples[t][i] for t in TRAITS}
        
        # Cấu trúc hội thoại đa thành phần chuẩn IFT
        messages = [
            {
                "role": "system", 
                "content": "You are an expert psychological assessment assistant. Analyze the given essay text and output the Big Five Personality Traits (cEXT, cNEU, cAGR, cCON, cOPN) strictly in a valid JSON format where 'y' indicates high and 'n' indicates low."
            },
            {
                "role": "user", 
                "content": f"Analyze the following essay text:\n\n{str(examples['text'][i])}"
            },
            {
                "role": "assistant", 
                "content": json.dumps(label)
            }
        ]
        
        # Áp dụng template Jinja nguyên bản của mô hình Qwen
        formatted_chat = tokenizer.apply_chat_template(messages, tokenize=False)
        texts.append(formatted_chat)
        
    return {"text": texts}

# Tiến hành ánh xạ (Map) cấu trúc hội thoại IFT vào Dataset
train_dataset = Dataset.from_pandas(train_df).map(format_ift, batched=True)
val_dataset   = Dataset.from_pandas(val_df).map(format_ift, batched=True)

print("\n=== Sample Formatted IFT Prompt ===")
print(train_dataset[0]["text"][:400])

# ============================================================
# 5. LOGGING CALLBACK
# ============================================================
training_logs = []

class LogCallback(TrainerCallback):
    def on_log(self, args, state, control, logs=None, **kwargs):
        if logs:
            entry = {"step": state.global_step}
            entry.update(logs)
            training_logs.append(entry)
            if "loss" in logs:
                print(f"  Step {state.global_step:4d} | loss={logs['loss']:.4f}"
                      + (f" | eval_loss={logs.get('eval_loss',''):.4f}"
                         if "eval_loss" in logs else ""))

# ============================================================
# 6. TRAINING PROCESS
# ============================================================
gc.collect()
torch.cuda.empty_cache()

trainer = SFTTrainer(
    model=model,
    processing_class=tokenizer,
    train_dataset=train_dataset,
    eval_dataset=val_dataset,
    callbacks=[LogCallback()],
    args=SFTConfig(
        output_dir=OUTPUT_DIR,
        dataset_text_field="text",
        max_length=MAX_LENGTH,
        per_device_train_batch_size=1,
        gradient_accumulation_steps=8,
        num_train_epochs=3,
        learning_rate=2e-4,
        warmup_ratio=0.05,
        fp16=False,
        bf16=False,
        optim="paged_adamw_8bit",
        lr_scheduler_type="cosine",
        weight_decay=0.01,
        logging_steps=20,
        eval_steps=100,
        save_steps=200,
        save_total_limit=2,
        gradient_checkpointing=False,
        report_to="none",
        seed=SEED,
    ),
)

print("\n🚀 Starting Instruction Fine-Tuning (IFT) training...")
start_time = time.time()
trainer.train()
train_time = time.time() - start_time
print(f"\n✅ IFT Training completed in {train_time/3600:.2f} hours")

# Lưu training logs với tên chuẩn hóa
logs_df = pd.DataFrame(training_logs)
logs_df.to_csv("/kaggle/working/training_logs.csv", index=False)
print("Saved training_logs.csv")

# Lưu trọng số của mô hình sau tinh chỉnh
model.save_pretrained(f"{OUTPUT_DIR}/final_model")
tokenizer.save_pretrained(f"{OUTPUT_DIR}/final_model")
print("Saved final_model")

# ============================================================
# 7. EVALUATION ON INDEPENDENT TEST SET (IFT Prompts)
# ============================================================
print("\n🔍 Evaluating model on independent test set...")
model.eval()

def predict_traits_ift(text):
    # Cấu trúc câu hỏi chuẩn ChatML đồng nhất với lúc huấn luyện
    messages = [
        {
            "role": "system", 
            "content": "You are an expert psychological assessment assistant. Analyze the given essay text and output the Big Five Personality Traits (cEXT, cNEU, cAGR, cCON, cOPN) strictly in a valid JSON format where 'y' indicates high and 'n' indicates low."
        },
        {
            "role": "user", 
            "content": f"Analyze the following essay text:\n\n{str(text)}"
        }
    ]
    
    # Sinh prompt kèm thẻ mồi kích hoạt sinh từ của Assistant (<|im_start|>assistant)
    prompt = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
    
    inputs = tokenizer(
        prompt, return_tensors="pt",
        truncation=True, max_length=MAX_LENGTH
    ).to(model.device)

    with torch.no_grad():
        outputs = model.generate(
            **inputs,
            max_new_tokens=60,
            do_sample=False,
            temperature=1.0,
            pad_token_id=tokenizer.eos_token_id,
        )

    # Lấy chính xác phần text mà mô hình sinh mới ra ở phía sau prompt gốc
    generated = tokenizer.decode(
        outputs[0][inputs["input_ids"].shape[1]:],
        skip_special_tokens=True
    ).strip()

    try:
        start  = generated.find("{")
        end    = generated.rfind("}") + 1
        parsed = json.loads(generated[start:end])
        return {t: parsed.get(t) for t in TRAITS}
    except:
        return {t: None for t in TRAITS}

# Chạy mô hình dự đoán trên toàn bộ Test Set
results = []
for i, (_, row) in enumerate(test_df.iterrows()):
    pred = predict_traits_ift(row["text"])
    results.append({
        **{f"true_{t}": row[t] for t in TRAITS},
        **{f"pred_{t}": pred[t] for t in TRAITS},
    })
    if (i + 1) % 50 == 0:
        print(f"  [{i+1}/{len(test_df)}] samples predicted")

# Đã đổi tên file xuất ra thành predictions_ift_essays.csv
results_df = pd.DataFrame(results)
results_df.to_csv("/kaggle/working/predictions_ift_essays.csv", index=False)

# ============================================================
# 8. DETAILED CLASSIFICATION REPORT & METRICS GENERATION
# ============================================================
print("\n" + "="*60)
print("🎯 EVALUATION RESULTS — IFT Essays (Qwen2.5-3B-Instruct)")
print("="*60)

summary = []
for t in TRAITS:
    true_vals = results_df[f"true_{t}"].map({"y": 1, "n": 0})
    pred_vals = results_df[f"pred_{t}"].map({"y": 1, "n": 0})
    valid     = true_vals.notna() & pred_vals.notna()
    n_valid   = valid.sum()

    if n_valid > 0:
        tv = true_vals[valid]
        pv = pred_vals[valid]
        acc  = accuracy_score(tv, pv)
        prec = precision_score(tv, pv, zero_division=0)
        rec  = recall_score(tv, pv, zero_division=0)
        f1   = f1_score(tv, pv, zero_division=0)
        f1m  = f1_score(tv, pv, average="macro", zero_division=0)

        print(f"\n🧠 Trait {t}:")
        print(f"  Accuracy:   {acc:.4f} ({acc:.2%})")
        print(f"  Precision:  {prec:.4f}")
        print(f"  Recall:     {rec:.4f}")
        print(f"  F1-Score:   {f1:.4f}")
        print(f"  F1-Macro:   {f1m:.4f}")
        print(f"  Valid Test Samples: {n_valid}")

        summary.append({
            "trait": t, "accuracy": round(acc,4),
            "precision": round(prec,4), "recall": round(rec,4),
            "f1": round(f1,4), "f1_macro": round(f1m,4),
            "n_valid": n_valid
        })

# Đã đổi tên file thống kê thành summary_ift_essays.csv
summary_df = pd.DataFrame(summary)
summary_df.to_csv("/kaggle/working/summary_ift_essays.csv", index=False)

print("\n" + "="*60)
print("📊 OVERALL AVERAGE METRICS ACROSS ALL 5 TRAITS")
print("="*60)
print(f"  Avg Accuracy:  {summary_df['accuracy'].mean():.4f}")
print(f"  Avg Precision: {summary_df['precision'].mean():.4f}")
print(f"  Avg Recall:    {summary_df['recall'].mean():.4f}")
print(f"  Avg F1-Score:  {summary_df['f1'].mean():.4f}")
print(f"  Avg F1-Macro:  {summary_df['f1_macro'].mean():.4f}")

print("\n" + "="*60)
print("📋 EXPERIMENT METADATA SUMMARY")
print("="*60)
print(f"  Fine-Tuning Method: Instruction Fine-Tuning (IFT)")
print(f"  Base Architecture:  {MODEL_NAME}")
print(f"  Hardware Compute:   {torch.cuda.get_device_name(0)}")
print(f"  Training Duration:  {train_time/3600:.2f} hours")
print(f"  Train Samples Count: {len(train_df)}")
print(f"  Test Samples Count:  {len(test_df)}")
print(f"  Epochs Configured:  3")
print(f"  LoRA Configuration: Rank={16}, Alpha={32}, Dropout=0.05")

print("\n✅ All clean IFT results successfully generated and saved!")
print("  - /kaggle/working/predictions_ift_essays.csv")
print("  - /kaggle/working/summary_ift_essays.csv")
print("  - /kaggle/working/training_logs.csv")
