df = pd.read_csv(DATA_PATH)
df = df[["TEXT_CLEAN"] + TRAITS].dropna().reset_index(drop=True)

for trait in TRAITS:
    df[trait] = df[trait].astype(str).str.strip().str.lower().map({"y": 1, "1": 1, "1.0": 1, "n": 0, "0": 0, "0.0": 0})
df = df.dropna(subset=TRAITS).reset_index(drop=True)

print(f"✅ Tải dữ liệu thành công:")

# ── CELL 4: CHUẨN HÓA CHAT TEMPLATE & CHE NHÃN PROMPT VÙNG LABELS ─────
tokenizer = AutoTokenizer.from_pretrained(MODEL_BASE, trust_remote_code=True)
if tokenizer.pad_token is None:
    tokenizer.pad_token = tokenizer.eos_token
tokenizer.padding_side = "right"

SYSTEM_PROMPT = (
    "You are an expert psychological assessment assistant. "
    "Analyze the given text and predict the MBTI personality traits. "
    "Output ONLY a valid JSON object with keys mEXT, mOPN, mAGR, mCON "
    "and values 'y' (high) or 'n' (low). No explanation needed."
)

def preprocess_df_to_dataset(input_df):
    tokenized_data = []
    for _, row in input_df.iterrows():
        label_dict = {t: ("y" if int(row[t]) == 1 else "n") for t in TRAITS}
        m_full   = [{"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": f"Text:\n{str(row['TEXT_CLEAN'])}"}, {"role": "assistant", "content": json.dumps(label_dict)}]
        m_prompt = [{"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": f"Text:\n{str(row['TEXT_CLEAN'])}"} ]
        
        full_text   = tokenizer.apply_chat_template(m_full, tokenize=False, add_generation_prompt=False)
        prompt_text = tokenizer.apply_chat_template(m_prompt, tokenize=False, add_generation_prompt=True)
        
        t_full   = tokenizer(full_text, truncation=True, max_length=MAX_LENGTH)
        t_prompt = tokenizer(prompt_text, truncation=True, max_length=MAX_LENGTH)
        
        input_ids = t_full["input_ids"]
        labels    = list(input_ids)
        p_len     = min(len(t_prompt["input_ids"]), len(input_ids))
        
        for i in range(p_len):
            labels[i] = -100 # Che vùng prompt
            
        tokenized_data.append({"input_ids": input_ids, "attention_mask": t_full["attention_mask"], "labels": labels})
    return Dataset.from_list(tokenized_data)

print("\n⚙️ Đang mã hóa dữ liệu văn bản theo cấu trúc Chat Template...")
train_dataset = preprocess_df_to_dataset(train_2b_df)
eval_dataset  = preprocess_df_to_dataset(val_df)

# ── CELL 5: TẢI BASE MODEL QUAN TIZA & NẠP ADAPTER TỪ CHẶNG 1 ─────────
print("\n" + "=" * 60)

print("=" * 60)

bnb_config = BitsAndBytesConfig(
    load_in_4bit=True, bnb_4bit_quant_type="nf4",
    bnb_4bit_use_double_quant=True, bnb_4bit_compute_dtype=torch.float16
)

base_model = AutoModelForCausalLM.from_pretrained(
    MODEL_BASE, quantization_config=bnb_config, device_map={"": 0}, trust_remote_code=True
)
base_model.config.use_cache = False
base_model = prepare_model_for_kbit_training(base_model)

# Đọc adapter đã học từ Chặng 1 để tiếp tục tối ưu hóa chuyên sâu
if os.path.exists(STAGE_1_MODEL_PATH):
    print(f"🚀 Tìm thấy Adapter Chặng 1 tại [{STAGE_1_MODEL_PATH}]. Đang nạp để học tiếp...")
    model = PeftModel.from_pretrained(base_model, STAGE_1_MODEL_PATH, is_trainable=True)
else:
    print(f"⚠️ Không tìm thấy Chặng 1 tại {STAGE_1_MODEL_PATH}. Khởi tạo Adapter mới hoàn toàn.")
    peft_config = LoraConfig(
        r=16, lora_alpha=32, target_modules=["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"],
        lora_dropout=0.05, bias="none", task_type="CAUSAL_LM"
    )
    model = get_peft_model(base_model, peft_config)

model.print_trainable_parameters()

# ── CELL 6: THIẾT LẬP CUSTOM COLLATOR VÀ TIẾN HÀNH FINE-TUNE ──────────
print("\n" + "=" * 60)
print("🔥 BƯỚC 3: PHÁT ĐỘNG QUÁ TRÌNH HUẤN LUYỆN CHẶNG 2")
print("=" * 60)

class CustomDataCollatorForIFT:
    def __init__(self, tokenizer): self.tokenizer = tokenizer
    def __call__(self, features):
        max_len = max(len(f["input_ids"]) for f in features)
        input_ids, attention_mask, labels = [], [], []
        for f in features:
            rem = max_len - len(f["input_ids"])
            input_ids.append(f["input_ids"] + [self.tokenizer.pad_token_id] * rem)
            attention_mask.append(f["attention_mask"] + [0] * rem)
            labels.append(f["labels"] + [-100] * rem)
        return {"input_ids": torch.tensor(input_ids, dtype=torch.long), "attention_mask": torch.tensor(attention_mask, dtype=torch.long), "labels": torch.tensor(labels, dtype=torch.long)}

training_logs = []
class LogCallback(TrainerCallback):
    def on_log(self, args, state, control, logs=None, **kwargs):
        if logs and ("loss" in logs or "eval_loss" in logs):
            training_logs.append({"step": state.global_step, "epoch": round(state.epoch, 4) if state.epoch else 0, "train_loss": logs.get("loss"), "eval_loss": logs.get("eval_loss")})
            if "loss" in logs: print(f"   [Step {state.global_step:4d}] Train Loss = {logs['loss']:.4f}")
            if "eval_loss" in logs: print(f"    📊 [EVAL] Step {state.global_step:4d} | Val Loss = {logs['eval_loss']:.4f}")

training_args = TrainingArguments(
    output_dir=OUTPUT_DIR, per_device_train_batch_size=1, gradient_accumulation_steps=8,
    per_device_eval_batch_size=1, learning_rate=2e-4, lr_scheduler_type="cosine",
    warmup_ratio=0.05, weight_decay=0.01, num_train_epochs=2, fp16=True,
    logging_steps=20, eval_strategy="steps", eval_steps=100, save_strategy="steps", save_steps=100,
    save_total_limit=2, load_best_model_at_end=True, metric_for_best_model="eval_loss",
    greater_is_better=False, gradient_checkpointing=True, report_to="none", seed=SEED
)

trainer = Trainer(
    model=model, args=training_args, train_dataset=train_dataset, eval_dataset=eval_dataset,
    data_collator=CustomDataCollatorForIFT(tokenizer), callbacks=[LogCallback()]
)

gc.collect()
torch.cuda.empty_cache()
trainer.train()

def parse_json_output(text):
    """Bóc tách chuỗi JSON dự đoán từ mô hình sinh chữ thành từ điển nhãn"""
    try:
        match = re.search(r"\{.*?\}", text.replace("\n", ""))
        if match:
            d = json.loads(match.group(0))
            return [1 if d.get(t, "n") == "y" else 0 for t in TRAITS]
    except: pass
    return [0, 0, 0, 0] # Mặc định trả về âm tính nếu chuỗi lỗi cấu trúc

all_probs, all_preds, all_labels = [], [], []
BATCH_SIZE = 4
total_batches = (len(test_df) + BATCH_SIZE - 1) // BATCH_SIZE

for idx in range(total_batches):
    sub = test_df.iloc[idx*BATCH_SIZE : min((idx+1)*BATCH_SIZE, len(test_df))]
    prompts = []
    for _, r in sub.iterrows():
        m = [{"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": f"Text:\n{str(r['TEXT_CLEAN'])}"}]
        prompts.append(tokenizer.apply_chat_template(m, tokenize=False, add_generation_prompt=True))
        
    inputs = tokenizer(prompts, return_tensors="pt", padding=True, truncation=True, max_length=MAX_LENGTH-100).to(model.device)
    with torch.no_grad():
        outputs = model.generate(**inputs, max_new_tokens=60, do_sample=False, pad_token_id=tokenizer.eos_token_id)
        
    for i, out in enumerate(outputs):
        gen_text = tokenizer.decode(out[inputs["input_ids"].shape[1]:], skip_special_tokens=True).strip()
        pred_bin = parse_json_output(gen_text)
        true_bin = [int(sub.iloc[i][t]) for t in TRAITS]
        
        all_preds.append(pred_bin)
        all_labels.append(true_bin)
        all_probs.append([0.9 if p==1 else 0.1 for p in pred_bin]) # Xác suất tượng trưng cho IFT

    if (idx + 1) % 15 == 0 or (idx + 1) == total_batches:
        print(f"   Xử lý mẫu: {min((idx+1)*BATCH_SIZE, len(test_df))}/{len(test_df)}")

all_preds, all_labels, all_probs = np.array(all_preds), np.array(all_labels), np.array(all_probs)

# ── CELL 8: TRÍCH XUẤT THỐNG KÊ CHI TIẾT (METRICS & CASE STUDIES) ──────

print("📊 BƯỚC 5: XUẤT FILE ĐÁNH GIÁ CHẤT LƯỢNG (METRICS & CASES)")

metrics_rows = []
for i, trait in enumerate(TRAITS):
    acc, pre, rec, f1 = accuracy_score(all_labels[:, i], all_preds[:, i]), precision_score(all_labels[:, i], all_preds[:, i], zero_division=0), recall_score(all_labels[:, i], all_preds[:, i], zero_division=0), f1_score(all_labels[:, i], all_preds[:, i], zero_division=0)
    metrics_rows.append({"model": MODEL_LABEL, "trait": trait, "accuracy": round(acc, 4), "precision": round(pre, 4), "recall": round(rec, 4), "f1": round(f1, 4)})
    print(f"   {trait:<5}: Acc={acc:.4f} | Pre={pre:.4f} | Rec={rec:.4f} | F1={f1:.4f}")

macro_f1 = np.mean([r["f1"] for r in metrics_rows])
metrics_rows.append({"model": MODEL_LABEL, "trait": "MACRO", "accuracy": round(np.mean([r["accuracy"] for r in metrics_rows]), 4), "precision": round(np.mean([r["precision"] for r in metrics_rows]), 4), "recall": round(np.mean([r["recall"] for r in metrics_rows]), 4), "f1": round(macro_f1, 4)})
metrics_df = pd.DataFrame(metrics_rows)
metrics_df.to_csv(f"{OUTPUT_DIR}/metrics_summary.csv", index=False)

pred_rows = []
for i, (_, row) in enumerate(test_df.iterrows()):
    entry = {"TEXT_CLEAN": str(row["TEXT_CLEAN"])[:500], "exact_match": int(all(all_labels[i, j] == all_preds[i, j] for j in range(4)))}
    for j, t in enumerate(TRAITS):
        entry[f"{t}_gt"], entry[f"{t}_pred"] = int(all_labels[i, j]), int(all_preds[i, j])
    pred_rows.append(entry)
pred_df = pd.DataFrame(pred_rows)
pred_df.to_csv(f"{OUTPUT_DIR}/test_predictions.csv", index=False)

exact_rate = pred_df["exact_match"].mean() * 100
print(f"   {"─"*50}\n   🔥 MACRO AVG F1: {macro_f1:.4f} | Exact Match Rate (Đúng cả 4/4 nhóm): {exact_rate:.1f}%")

    plt.axvline(x=df_1a["step"].max(), color="green", linestyle=":", alpha=0.8)

plt.show()
